\begin{itemize}
    \item \textbf{M1:} Team formation, area identification, background, motivation, problem formulation
    \item \textbf{M2:} Mathematical modeling
    \begin{itemize}
        \item Random Variables (RV)
        \item PMF / PDF

        \item Multivariate RVs

    \end{itemize}

    \item \textbf{M4:} Inference using randomized algorithms
    \item \textbf{M5:} Randomized algorithms and comparison with deterministic algorithms
    \item \textbf{M6:} Bounds, analysis, final submission
\end{itemize}
